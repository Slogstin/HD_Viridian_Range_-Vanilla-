class BallisticsGelBasic : hdmobbase
{
	default
	{
		health 100;
		height 54;
		radius 12;
		deathheight 12;
		+hdmobbase.smallhead
		+hdmobbase.biped
		mass 10000;
		+dontthrust;
		+ismonster;
		translation "ballgelwhite";
		bloodcolor "f0 f0 f0";
		tag "Basic ballistics gel";
	}
	
	states
	{
		spawn:
			TROO A -1;
			stop;
			
		death:
			TROO NOPQRST 4;
			TROO U -1;
			stop;
			
		death.massacre:
			TROO A -1;
			stop;
			
		raise:
			TROO A 1
			{
				setstatelabel("spawn");
				spawn("telefog", pos);
				bodydamage = 0;
			}
	}
}
